# List databases
sudo -u postgres psql -l

# Connect to a database (e.g. rundeck)
sudo -u postgres psql rundeck

# Quit psql
\q

# List tables
\dt

# Describe table
\d+ example_table

# Make output verticle instead of horizonal (similar to MySQL \G)
\x

# Show users
\du

# Show default privileges
\ddp

# Show privileges
\dp

# Show databases
\l

# Show replication status
SELECT client_addr, application_name, state, sent_lsn, replay_lsn, pg_wal_lsn_diff(sent_lsn, replay_lsn) AS replay_lag_bytes, write_lag, flush_lag, replay_lag FROM pg_stat_replication;
